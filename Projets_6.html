Roche Papier Ciseaux

Cours : Programmation (1ere session TI)
Programme qui joue à roche papier ciseaux contre vous 

namespace RochePapierCiseaux_Exam2
{
    internal class Program
    {
        static void Main(string[] args)
        {
            rochePapierCiseauJeu();
        }

        static void rochePapierCiseauJeu()
        {
            Random random = new Random();
            int choixVariable = 0;
            int variableAleatoire = random.Next(1, 4);

            bool dead = false;
            int joueurGagne = 0;
            int ordiGagne = 0;

            while (!dead && joueurGagne < 3 && ordiGagne < 3)
            {
                Console.WriteLine("Veuillez choisir entre roche(1), papier(2), ciseau(3).");
                int.TryParse(Console.ReadLine(), out choixVariable);

                if (joueurGagne == 3 || ordiGagne == 3)
                {
                    dead = true;
                }

                // ICI LE JOUEUR GAGNE
                else if (variableAleatoire == 1 && choixVariable == 2) // Joueur qui gange
                {
                    // C'est le papier(2) qui gagne
                    Console.WriteLine("C'est le papier(2) qui gagne. Joueur qui gagne");
                    joueurGagne += 1;
                }
                else if (variableAleatoire == 2 && choixVariable == 3)
                {
                    // C'est le joueur qui gagne, les ciseaux(3) gagnent
                    Console.WriteLine("Ce sont les ciseaux(3) qui gagnent. Joueur qui gagne");
                    joueurGagne += 1;
                }
                else if (variableAleatoire == 3 && choixVariable == 1)
                {
                    // C'est la roche(1) qui gagne, joueur qui gagne
                    Console.WriteLine("C'est la roche(1) qui gagne, joueur qui gagne");
                    joueurGagne += 1;
                }

                // ICI L'ORDIIIIII GAGNE
                else if (variableAleatoire == 2 && choixVariable == 1)
                {
                    // C'est le papier(2) qui gagne, ordi gagne
                    Console.WriteLine("C'est le papier(2) qui gagne. Ordi qui gagne");
                    ordiGagne += 1;
                }
                else if (variableAleatoire == 3 && choixVariable == 2)
                {
                    // C'est l'ordi qui gagne, les ciseaux(3) gagnent
                    Console.WriteLine("Ce sont les ciseaux(3) qui gagnent. Ordi qui gagne");
                    ordiGagne += 1;
                }
                else if (variableAleatoire == 1 && choixVariable == 3)
                {
                    // C'est la roche(1) qui gagne, ordi qui gagne
                    Console.WriteLine("C'est la roche(1) qui gagne, Ordi qui gagne");
                    ordiGagne += 1;
                }
                else if (variableAleatoire == 1 && choixVariable == 1 || variableAleatoire == 2 && choixVariable == 2 || variableAleatoire == 3 && choixVariable == 3)
                {
                    Console.WriteLine("On recommence le tour, c'est une égalité");
                }
            }

            if (ordiGagne == 3)
            {
                Console.WriteLine("Ordi qui gagne");
            }
            else if (joueurGagne == 3)
            {
                Console.WriteLine("Joueur qui gagne");
            }
        }
    }
}
