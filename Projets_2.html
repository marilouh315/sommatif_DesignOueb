Akinator

Cours : Programmation (1ere session TI)
Exercice de programmation « Akinator », qui tentera de deviner un nombre. 

namespace Labo_Genie
{
    internal class Program
    {
        static void Main(string[] args)
        {
            // Demander à l'utilisateur s'il veut rejouer 
            //sans être obligé d'aller dans le programme et écrire des while et des if...
            
            bool jouer = true;
            string choixJoueur = "";
            
            while (jouer)
            {
                // Méthode du jeu (code plus bas)
                devine2();
                Console.Clear();
                Console.WriteLine("Rejouer? o/n");
                choixJoueur = Console.ReadLine();

                if (choixJoueur =="n")
                {
                    jouer = false;
                }
            }
        }

        static void devine2()
        {
            bool vrai = true;
            string reponse = "";
            int min = 1;
            int max = 100;
            int nombredevine = 0;

            Console.WriteLine("Pensez à un nombre entre 1 et 100, je vais tenter de deviner votre nombre...");
            Console.WriteLine("Appuyer pour continuer");
            Console.ReadKey();

            Console.WriteLine("Votre nombre est 50?");
            Console.WriteLine("Oui (o), plus grand que 50 (+), plus petit que 50 (-)");
            reponse = Console.ReadLine();

            while (vrai)
            {
                if (reponse == "+")
                {
                    min = 51;
                    nombredevine = (min + max) / 2;
                    while (vrai)
                    {
                        Console.WriteLine("Votre nombre est " + nombredevine + "?");
                        Console.WriteLine("Oui (o), plus grand que " + nombredevine + " (+), plus petit que " + nombredevine + " (-)");
                        reponse = Console.ReadLine();

                        if (reponse == "+")
                        {
                            min = nombredevine;
                            nombredevine = (min + max) / 2;
                        }
                        else if (reponse == "-")
                        {
                            max = nombredevine;
                            nombredevine = (min + max) / 2;
                        }
                        if (reponse == "o")
                        {
                            Console.WriteLine("J'ai réussi! Yay, j'ai deviné votre nombre à " + nombredevine);
                            vrai = false;
                            Console.ReadKey();
                        }
                    }
                }
                else if (reponse == "-")
                {
                    // Entre 1 et 50 (pour ne pas avoir des chiffres à virgules)
                    max = 49;
                    nombredevine = (min + max) / 2;

                    while (vrai)
                    {
                        Console.WriteLine("Votre nombre est " + nombredevine + "?");
                        Console.WriteLine("Oui (o), plus grand que " + nombredevine + " (+), plus petit que " + nombredevine + " (-)");
                        reponse = Console.ReadLine();

                        if (reponse == "o")
                        {
                            Console.WriteLine("J'ai réussi! Yay, j'ai deviné votre nombre à " + nombredevine);
                            vrai = false;
                            Console.ReadKey();
                        }
                        else if (reponse == "+")
                        {
                            min = nombredevine;
                            nombredevine = (min + max) / 2;
                        }
                        else if (reponse == "-")
                        {
                            max = nombredevine;
                            nombredevine = (min + max) / 2;
                        }
                    }
                }
                else if (reponse == "o")
                {
                    nombredevine = 50;
                    vrai = false;
                    Console.WriteLine("J'ai réussi! Yay, j'ai deviné votre nombre à " + nombredevine);
                    Console.ReadKey();
                }
            }
        }
    }
}
